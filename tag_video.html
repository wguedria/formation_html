Affichage d'un texte pour les navigateurs qui ne supportent pas canvas.

SOLUTION

myCanvas = document.getElementById('myCanvasId');
myCanvasContext = myCanvas.getContext('2d');
myCanvasContext.clearRect(0, 0, myCanvas.width, myCanvas.height);
myCanvasContext.moveTo(100, 150);
myCanvasContext.lineTo(450, 50);
myCanvasContext.stroke();
